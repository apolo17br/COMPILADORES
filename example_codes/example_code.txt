# Soma dois inteiros
@add ($a: int, $b: int) -> int {
  return $a + $b
}

# Subtrai dois inteiros
@sub ($a: int, $b: int) -> int {
  return $a - $b
}

# Ponto de entrada da aplicação
@main ($arg: char) -> int {
  if($arg == '+') {
    return @add(1, 1)
  } else {
    return @sub(1, 1)
  }
}
